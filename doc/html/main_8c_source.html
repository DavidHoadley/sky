4  * main.c - Simple demo program for Sun position using rectangular coordinates
5  *
6  * \author David Hoadley <vcrumble@westnet.com.au>
7  *
8  * \details
9  * This program calls two demo routines to calculate the Sun's position.
10  * The first calculates it using the NREL SPA algorithm (see reference),
11  * but implemented in rectangular coordinates. The second runs a loop to
12  * repeatedly calculate the position, but using interpolation between two
13  * previously calculated positions.
14  *
15  * The interpolation will introduce a very small error in position. For the
16  * value of 720 minutes supplied to routine skyfast_init(), the maximum
17  * error that is introduced is less than 0.075 arcseconds.
18  * Given that the NREL-SPA algorithm itself is only
19  * accurate to approximately 1 arcsecond, this is pretty good.
20  *
21  * This demo shows some of the use of the routines in sky.h and sun.h,
22  * and for interpolation, skyfast.h
23  *
24  * \par Reference:
25  * Reda, Ibrahim and Andreas, Afshin.
26  * _Solar Position Algorithm for Solar Radiation Applications._
27  * National Renewable Energy Laboratory, publication no.
28  * NREL/TP-560-34302, June 2003, revised January 2008
29  *
30  *==============================================================================
31  */

74 void sunTopocentricFast(double j2kUtc_d,
75  const Sky_DeltaTs *dTs,
76  const Sky_SiteProp *site,
77  Sky_SiteHorizon *topo)
78 /*! Uses previously calculated values of the Sun's apparent coordinates and
79  interpolation to quickly calculate the Sun's topocentric position.
80  \param[in] j2kUtc_d Days since Greenwich noon, 1-Jan-2000, UTC timescale
81  \param[in] deltas Delta T values, as set by the sky_initTime() (or
82  sky_initTimeSimple() or sky_initTimeDetailed()) routines
83  \param[in] site Properties of the observing site, particularly its
84  geometric location on the surface of the Earth, as set by
85  the sky_setSiteLocation() function (or sky_setSiteLoc2())
86  \param[out] topo Topocentric position, in both rectangular (unit vector)
87  form, and as Azimuth and Elevation (altitude) angles
88 
89  \par When to call this function
90  This function is designed to be called at high frequency (more than once
91  per second, even up to, say, 20 Hz).
92  \par
93  Use this function when you have made a previous call to skyfast_init().
94  Depending on how long a period you want to track the sun using this
95  routine, you may also need to run the function skyfast_backgroundUpdate() as
96  a low frequency, low priority routine, to update the interpolation ranges.
97  If you have not done this, call sun_nrelTopocentric() instead.
98 - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - */
99 {
100  Sky_Times atime;
101  Sky_TrueEquatorial approx;
102  V3D_Vector terInterV;// unit vector in Terrestrial Intermed Ref Sys
103 
104  REQUIRE_NOT_NULL(dTs);
105  REQUIRE_NOT_NULL(site);
106  REQUIRE_NOT_NULL(topo);
107 
108  sky_updateTimes(j2kUtc_d, dTs, &atime);
109 
110  /* Get the approximate apparent coordinates by interpolation */
111  skyfast_getApprox(atime.j2kTT_cy, &approx);
112 
113  /* Convert apparent coordinates to topocentric coordinates at the site */
114  sky0_appToTirs(&approx.appCirsV,
115  atime.j2kUT1_d,
116  approx.eqEq_rad,
117  &terInterV);
118  sky_siteTirsToTopo(&terInterV, approx.distance_au, site, topo);
119 }

123 void demo1(void)
124 /* Sample code demonstrating the calculation of the position of the Sun using
125  * rectangular coordinates
126 - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - */
127 {
128  Sky_DeltaTs deltaTs; /* Differences between timescales */
129  Sky_SiteProp site; /* Properties of the site */
130  Sky_SiteHorizon topo; /* Sun position, as seen from the site */
131  double j2kUtc_d; /* Time, expressed as a count of days */
132  char azStr[24], elStr[24];
133 
134  /* Set up the various delta T values */
135  sky_initTime(37, 0, &deltaTs);
136 
137  /* Set up the fixed site data */
138  sky_setSiteLocation(SITE_LATITUDE_deg,
139  SITE_LONGITUDE_deg,
140  SITE_HEIGHT_m,
141  &site);
142  sky_setSiteTempPress(SITE_TEMPERATURE_degC,
143  SITE_PRESSURE_hPa,
144  &site);
145  sky_setSiteTimeZone(SITE_TIMEZONE_h, &site);
146 
147  /* Get the current time in days since 2000-01-01 Greenwich noon */
148  j2kUtc_d = sky_unixTimeToJ2kd(time(NULL));
149 
150  /* Get sun position */
151  sun_nrelTopocentric(j2kUtc_d, &deltaTs, &site, &topo);
152 
153  /* Print out time and place */
154  printf("Full sun calculation\n");
155  skyio_printJ2kd(j2kUtc_d + site.timeZone_d);
156  skyio_radToDmsStr(azStr, sizeof(azStr), topo.azimuth_rad, 2);
157  skyio_radToDmsStr(elStr, sizeof(elStr), topo.elevation_rad, 2);
158  printf(" Sun Azimuth: %s, Elevation: %s\n", azStr, elStr);
159 }

163 void demo2(void)
164 /* Sample code using the greatly simplified calculation of the position
165  * of the Sun using interpolated coordinates.
166 - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - */
167 {
168  Sky_DeltaTs deltaTs; /* Differences between timescales */
169  Sky_SiteProp site; /* Properties of the site */
170  Sky_SiteHorizon topo; /* Sun position, as seen from the site */
171  double j2kUtc_d; /* Time, expressed as a count of days */
172  char azStr[24], elStr[24];
173  int i;
174 
175  /* Set up the various delta T values. Values are taken from Bulletin A of
176  * 16-Jul-2020, using the approximate "DUT1" value for deltaUT_s */
177  sky_initTime(37, -0.2, &deltaTs);
178 
179  /* Set up the fixed site data */
180  sky_setSiteLocation(SITE_LATITUDE_deg,
181  SITE_LONGITUDE_deg,
182  SITE_HEIGHT_m,
183  &site);
184  sky_setSiteTempPress(SITE_TEMPERATURE_degC,
185  SITE_PRESSURE_hPa,
186  &site);
187  sky_setSiteTimeZone(SITE_TIMEZONE_h, &site);
188 
189  /* Get the current time in days since 2000-01-01 Greenwich noon */
190  j2kUtc_d = sky_unixTimeToJ2kd(time(NULL));
191 
192  /* Set up the interpolation algorithm to calculate Sun positions */
193  skyfast_init(j2kUtc_d, 720, &deltaTs, &sun_nrelApparent);
194 
195  printf("Fast sun calculation\n");
196  for (i = 0; i < 100; i++) {
197  /* Get sun position using interpolation */
198  j2kUtc_d = sky_unixTimeToJ2kd(time(NULL));
199  sunTopocentricFast(j2kUtc_d, &deltaTs, &site, &topo);
200 
201  skyio_printJ2kd(j2kUtc_d + site.timeZone_d);
202  skyio_radToDmsStr(azStr, sizeof(azStr), topo.azimuth_rad, 2);
203  skyio_radToDmsStr(elStr, sizeof(elStr), topo.elevation_rad, 2);
204  printf(" Sun Azimuth: %s, Elevation: %s\n", azStr, elStr);
205 
206  sleep(1); /* Just use a simple delay for this demo. */
207  }
