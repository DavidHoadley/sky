70 /* Note there is some inconsistency in the use of terms here, when compared
71  * with the fundamental arguments of the 1980 nutation theory.
72  * NREL Moon Nutation1980 Term description
73  * L' (L) Mean Longitude of the Moon
74  * D D Mean Elongation of the Moon from the Sun
75  * M l' Mean Anomaly of the Sun
76  * M' l Mean Anomaly of the Moon
77  * Ω Longitude of ascending node of Moon
78  * F F = L - Ω, Mean Longitude of Moon minus Longitude
79  * of ascending node.
80  *
81  * In the NREL SAMPA document, sections 3.2.5 and 3.2.4, the term F is referred
82  * to as the "Argument of Latitude". I believe this is wrong. F is actually
83  * the MEAN Longitude of Moon minus the Longitude of the Moon's ascending node
84  * (F = L - Ω), or alternatively the MEAN anomaly plus the argument of periapsis
85  * (F = M' + ω). But the "Argument of Latitude" is defined as the TRUE anomaly
86  * plus the argument of periapsis - i.e. u = ν + ω.
87  *
88  * In the code that follows, I have kept this label, to make it easier for
89  * someone reading the NREL SAMPA document to follow the code, but I have marked
90  * it as "so-called" each time.
91  */

145 GLOBAL void moon_nrelApp2(double t_cy,
146  const Sky0_Nut1980 *nut,
147  V3D_Vector *appV,
148  double *dist_au)
149 /*! Calculates the Moon's position in geocentric apparent coordinates, using the
150  NREL Moon Position Algorithm.
151  \param[in] t_cy Julian centuries since J2000.0, TT timescale
152  \param[in] nut Nutation terms and obliquity of the ecliptic
153  \param[in] epsM Rotation matrix from ecliptic frame to equatorial frame,
154  as returned by function astc0_createEps1980Matrix(). It
155  is equal to R1(-ε) where ε is the true obliquity of
156  the ecliptic
157  \param[out] appV Position vector of Moon in apparent coordinates (unit
158  vector i.e. direction cosines)
159  \param[out] dist_au Geocentric distance of the Moon (Astronomical Units)
160 
161  \par Reference
162  Reda, I., "Solar Eclipse Monitoring for Solar Energy Applications Using the
163  Solar and Moon Position Algorithms". National Renewable Energy Laboratory
164  Technical Report NREL/TP-3B0-47681, March 2010
165 - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - */
166 {
167  OrbTerms orbt; // Orbital terms for the Moon
168  double a1_rad, a2_rad, a3_rad; // mysterious factors (radian)
169  double l_udeg, b_udeg; // Moon long & lat terms (micro-degrees)
170  double r_m; // Moon distance term (metres)
171  double dl_udeg, db_udeg; // Δl & Δb (micro-degrees)
172  double lamdap_rad; // λ' - Moon uncorrected longitude (radian)
173  double lamda_rad; // λ - Moon apparent longitude (radian)
174  double beta_rad; // β - Moon latitude (radian)
175  double mDelta_km; // Δ - Moon geocentric distance (km)
176  V3D_Matrix epsM; // Rotation by obliquity of ecliptic
177  V3D_Vector eclV; // Rectangular coordinates in ecliptic frame
178 
179  REQUIRE_NOT_NULL(appV);
180  REQUIRE_NOT_NULL(dist_au);
181 
182  /* Steps 3.2.1 tp 3.2.5, and equation (15) of 3.2.6 */
183  moonOrbitals(t_cy, &orbt);
184 
185  /* Steps 3.2.6 to 3.2.8 (the first!) */
186  moonLongDist(&orbt, &l_udeg, &r_m);
187  b_udeg = moonLatitude(&orbt);
188 
189  /* Steps 3.2.8 (the second!) 3.2.9 and 3.2.10 */
190  a1_rad = degToRad(119.75 + 131.849 * t_cy);
191  a2_rad = degToRad(53.09 + 479264.29 * t_cy);
192  a3_rad = degToRad(313.45 + 481266.484 * t_cy);
193 
194  /* Steps 3.2.11 & 3.2.12 */
195  dl_udeg = 3958.0 * sin(a1_rad) + 1962.0 * sin(orbt.lp_rad - orbt.f_rad)
196  + 318.0 * sin(a2_rad);
197  db_udeg = -2235.0 * sin(orbt.lp_rad) + 382.0 * sin(a3_rad)
198  + 175.0 * sin(a1_rad - orbt.f_rad)
199  + 175.0 * sin(a1_rad + orbt.f_rad)
200  + 127.0 * sin(orbt.lp_rad - orbt.mp_rad)
201  - 115.0 * sin(orbt.lp_rad + orbt.mp_rad);
202 
203  /* Steps 3.2.13 to 3.2.16 */
204  lamdap_rad = normalize(orbt.lp_rad + degToRad((l_udeg + dl_udeg) / 1e6),
205  TWOPI);
206  beta_rad = degToRad((b_udeg + db_udeg) / 1e6);
207  mDelta_km = 385000.56 + r_m / 1000.0;
208  /* Convert distance to the Moon from km to Astronomical Units, because all
209  other celestial objects are specified that way, so our routines
210  converting geocentric to topocentric place expect it to be that way. */
211  *dist_au = mDelta_km / au_km;
212 
213  /* Step 3.6 */
214  lamda_rad = lamdap_rad + nut->dPsi_rad;
215 
216  /* Convert to rectangular coordinates in ecliptic plane and rotate to
217  equatorial plane (i.e to RA/Dec frame). */
218  v3d_polarToRect(&eclV, lamda_rad, beta_rad);
219  v3d_createRotationMatrix(&epsM, Xaxis, -(nut->eps0_rad + nut->dEps_rad));
220  v3d_multMxV(appV, &epsM, &eclV);
221 
222  /* Now have vector of apparent coords. Could convert to RA and Dec using
223  v3d_rectToPolar(&RA, &Dec, appV) but there isn't really any need.
224  Had we done so we would have performed steps 3.8 and 3.9 of NREL SAMPA */
225 #if 0
226  printf("Ecliptic: Apparent Longitude = %12.7f, "
227  "Latitude = %f, Distance = %7.3f\n",
228  radToDeg(lamda_rad), radToDeg(beta_rad), mDelta_km / 6378.137);
229 #endif
230 #if 0
231  {
232  char astr1[24], astr2[24];
233  double RA, Dec;
234 
235  v3d_rectToPolar(&RA, &Dec, appV);
236  RA = normalize(RA, TWOPI);
237  printf("My Apparent* RA %f [%s], Dec %f [%s]\n",
238  radToDeg(RA),
239  fmtHms(astr1, sizeof(astr1), RA, 1),
240  radToDeg(Dec),
241  fmtDms(astr2, sizeof(astr2), Dec, 0));
242  }
243 #endif
244 }

248 GLOBAL void moon_nrelApparent(double j2kTT_cy, Sky_TrueEquatorial *pos)
249 /*! Calculate the Moon's position as a unit vector and a distance, in apparent
250  coordinates. It calls moon_nrelApp2() to obtain the Moon's position, after
251  having called sky0_nutationSpa() to obtain the necessary nutation terms.
252  \param[in] j2kTT_cy Julian centuries since J2000.0, TT timescale
253  \param[out] pos Timestamped structure containing position data and the
254  equation of the equinoxes
255 
256  \par When to call this function
257  Because this function is computationally intensive, you may wish to limit
258  your use of this function.
259  - if you want the Moon's position at multiple sites simultaneously at a
260  single time, call this function, then follow it with a call to routine
261  sky0_appToTirs(), and then make a separate call to
262  sky_siteTirsToTopo() for each of one those sites.
263  - if you want the Moon's position at one or more sites at closely spaced
264  times (e.g. for tracking the Moon), pass this function to the
265  skyfast_init() function. skyfast_init() will
266  call it for you several times, to fully calculate positions that will
267  be saved and used later for interpolation by the skyfast_getApprox()
268  function for tracking.
269  \par
270  \em Alternatives:
271  - If you want the Moon's position at a single site only at a single
272  time, call moon_nrelTopocentric() instead, and it will call this
273  function for you.
274  - If you want the Moon's position at a single site for more than one
275  time but the times are spaced more than an hour or so apart, once
276  again call moon_nrelTopocentric() instead.
277  - If you want to track the Moon, call skyfast_getApprox() instead.
278  But this requires you to set up interpolation first with function
279  skyfast_init() (and this function), as described above.
280 - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - */
281 {
282  Sky0_Nut1980 nut;
283 
284  REQUIRE_NOT_NULL(pos);
285 
286  /* Calculate nutation (steps 3.4 of the algorithm in the SPA document) */
287  sky0_nutationSpa(j2kTT_cy, &nut);
288 
289  /* Calculate the mean obliquity of the ecliptic (step 3.5) and the equation
290  of the equinoxes */
291  sky0_epsilonSpa(j2kTT_cy, &nut);
292  pos->eqEq_rad = nut.eqEq_rad;
293 
294  /* Calculate Moon apparent position */
295  moon_nrelApp2(j2kTT_cy, &nut, &pos->appCirsV, &pos->distance_au);
296 
297  /* Now set the timestamp*/
298  pos->timestamp_cy = j2kTT_cy;
299 }

303 GLOBAL void moon_nrelTopocentric(double j2kdUtc,
304  const Sky_DeltaTs *deltas,
305  const Sky_SiteProp *site,
306  Sky_SiteHorizon *topo)
307 /*! Calls moon_nrelApparent() to calculate the Moon's position in apparent
308  * coordinates using the NREL Moon Position Algorithm, and then converts this
309  * to topocentric horizon coordinates at the specified site.
310  \param[in] j2kdUtc UTC time in "J2KD" form - i.e days since J2000.0
311  (= JD - 2 451 545.0)
312  \param[in] deltas Delta T values, as set by the sky_initTime() (or
313  sky_initTimeSimple() or sky_initTimeDetailed()) routines
314  \param[in] site Properties of the observing site, particularly its
315  geometric location on the surface of the Earth, as set by
316  the sky_setSiteLocation() function (or sky_setSiteLoc2())
317  \param[out] topo Topocentric position, in both rectangular (unit vector)
318  form, and as Azimuth and Elevation (altitude) angles
319 - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - */
320 {
321  Sky_Times atime; // time, in various timescales
322  Sky_TrueEquatorial pos; // geocentric position of the Moon and dist.
323  V3D_Vector terInterV; // unit vector in Terrestrial Intermed Ref Sys
324 
325  REQUIRE_NOT_NULL(deltas);
326  REQUIRE_NOT_NULL(site);
327  REQUIRE_NOT_NULL(topo);
328 
329  sky_updateTimes(j2kdUtc, deltas, &atime);
330 
331  /* Calculate Moon apparent position */
332  moon_nrelApparent(atime.j2kTT_cy, &pos);
333 
334  /* Convert apparent position to topocentric Azimuth/Elevation coords */
335  sky0_appToTirs(&pos.appCirsV, atime.j2kUT1_d, pos.eqEq_rad, &terInterV);
336  sky_siteTirsToTopo(&terInterV, pos.distance_au, site, topo);
337 }

341 GLOBAL double moon_riseSet(int year,
342  int month,
343  int day,
344  bool getMoonrise,
345  const Sky_DeltaTs *deltas,
346  const Sky_SiteProp *site,
347  Sky_SiteHorizon *topo)
348 /*! Routine to calculate the time of moonrise or moonset for the day specified
349  by \a year, \a month and \a day. This function uses the NREL MPA algorithm
350  of moon_nrelTopocentric() to calculate the Moon's position.
351  \returns Moonrise (or moonset) time for the day given in
352  \a year, \a month and \a day (returned as a J2KD date
353  (= JD - 2 451 545.0), UTC timescale).
354  To view this as a local date and time, add this
355  value to \a site->timezone_d and pass the result to
356  function sky_j2kdToCalTime().
357  \param[in] year, month, day
358  Date for which moonrise or moonset time is desired
359  \param[in] getMoonrise If true, get moonrise time. If false, get moonset time
360  \param[in] deltas Delta T values, as set by the sky_initTime() (or
361  sky_initTimeSimple() or sky_initTimeDetailed())
362  routines
363  \param[in] site Properties of the observing site, particularly its
364  geometric location on the surface of the Earth and its
365  time zone, as set by the sky_setSiteLocation() function
366  (or sky_setSiteLoc2()) and sky_setSiteTimeZone().
367  \param[out] topo \b Optional. Topocentric position of the Moon at rise
368  or set, in both rectangular (unit vector) form, and as
369  Azimuth and Elevation (altitude) angles. If you are not
370  interested in these values, you can pass NULL to this
371  parameter.
372 
373  This routine uses an iterative approach. Three iterations seems to be enough.
374 
375  \note
376  If the Moon does not rise (or set) on the specified day, this routine will
377  return the time just before midnight on the previous day, or just after
378  midnight on the next day, when the rise (or set) actually occurs.
379  If the routine encounters an error, it will return 0.0
380 - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - */
381 {
382  Sky_SiteHorizon topo1; // Moon apparent position
383  double estimate_d; // estimate of the MJD of Moon rise or set time
384  Sky_SiteProp st; // copy of site details, without refraction
385  int i; // iteration counter
386 
387  REQUIRE_NOT_NULL(deltas);
388  REQUIRE_NOT_NULL(site);
389 
390  /* Moonrise and set calculations assume a standard refraction at the horizon
391  * of 34 arcminutes, and a Moon semi-diameter of 16 arcminutes. So the
392  * calculation is based on the time at which the UNREFRACTED Moon position is
393  * at -50 arcmin. So copy site information, and set refraction to zero for
394  * that copy, which will be passed to riseSetApprox() */
395  st = *site;
396  st.refracPT = 0.0;
397 
398  /* Make an initial guess of civil noon on the specified day */
399  estimate_d = sky_calTimeToJ2kd(year, month, day,
400  12, 0, 0.0, site->timeZone_d * 24.0);
401 
402  /* Iterate to settle on a time. */
403  for (i = 0; i < 3; i++) {
404  estimate_d = riseSetApprox(estimate_d, getMoonrise, deltas,&st, &topo1);
405  if (estimate_d == 0.0) {
406  /* Error in calculations on the requested day.
407  Don't try another iteration. */
408  break;
409  }
410 #if 0
411  printf("Moon rise/set estimate %d: ", i + 1);
412  printMjd(estimate_d + site->timeZone_d);
413  printf("\n");
414 #endif
415  }
416 
417  if (topo != NULL) {
418  *topo = topo1;
419  }
420  return estimate_d;
421 }

430 LOCAL void moonOrbitals(double t_cy, OrbTerms *orb)
431 /* Calculate the fundamental orbital parameters required to get the moon
432  position. This is steps 3.2.1 to 3.2.5 of the NREL SAMPA document.
433  Inputs
434  t_cy - julian centuries since J2000.0, TT timescale
435  Outputs
436  orb - orbital terms for the moon, and eccentricity of Earth's orbit
437 - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - */
438 {
439  double temp_deg;
440 
441  /* Moon's Mean Longitude L' */
442  temp_deg = 218.3164477
443  + t_cy * (481267.88123421
444  + t_cy * (-0.0015786
445  + t_cy * (1.0 / 538841.0
446  + t_cy * (-1.0 / 65194000.0))));
447  orb->lp_rad = normalize(degToRad(temp_deg), TWOPI);
448 
449  /* Mean Elongation of the Moon D */
450  temp_deg = 297.8501921
451  + t_cy * (445267.1114034
452  + t_cy * (-0.0018819
453  + t_cy * (1.0 / 545868.0
454  + t_cy * (-1.0 / 113065000.0))));
455  orb->d_rad = normalize(degToRad(temp_deg), TWOPI);
456 
457  /* Sun's Mean Anomaly M */
458  temp_deg = 357.5291092
459  + t_cy * (35999.0502909
460  + t_cy * (-0.0001536
461  + t_cy * (1.0 / 24490000.0)));
462  orb->m_rad = normalize(degToRad(temp_deg), TWOPI);
463 
464  /* Moon's Mean Anomaly M' */
465  temp_deg = 134.9633964
466  + t_cy * (477198.8675055
467  + t_cy * (0.0087414
468  + t_cy * (1.0 / 69699.0
469  + t_cy * (-1.0 / 14712000.0))));
470  orb->mp_rad = normalize(degToRad(temp_deg), TWOPI);
471 
472  /* Moon's so-called Argument of Latitude F */
473  temp_deg = 93.2720950
474  + t_cy * (483202.0175233
475  + t_cy * (-0.0036539
476  + t_cy * (-1.0 / 3526000.0
477  + t_cy * (1.0 / 863310000.0))));
478  orb->f_rad = normalize(degToRad(temp_deg), TWOPI);
479 
480  /* Eccentricity of the Earth's orbit around the Sun */
481  orb->e = 1.0 - 0.002516 * t_cy - 0.0000074 * t_cy * t_cy;
482 #if 0
483  ///+
484  printf("Orbitals L' %12.7f, D %12.7f, M %12.6f, "
485  "M' %12.6f, F %12.7f, E %12.7f\n",
486  radToDeg(orb->lp_rad), radToDeg(orb->d_rad), radToDeg(orb->m_rad),
487  radToDeg(orb->mp_rad), radToDeg(orb->f_rad), orb->e);
488  ///-
489 #endif
490 }

697 LOCAL double riseSetApprox(double risesetGuess_d,
698  bool getMoonrise,
699  const Sky_DeltaTs *deltas,
700  const Sky_SiteProp *site,
701  Sky_SiteHorizon *topo)
702 /* Routine to calculate the time of Moon rise or set for the day specified by
703  MJDrisesetGuess. The result returned is an approximate value, whose accuracy
704  depends upon how close MJDrisesetGuess is to true Moon rise or set time.
705 
706  Of course, if this routine is called again with the previous result used as
707  the new guess, the new result will be more accurate.
708  Returns - Improved estimate of time of moonrise or moonset, (or zero
709  if the moon does not rise or does not set on this date)
710  Inputs
711  risesetGuess_d - MJD of date at which moonrise or moonset time is desired,
712  and time of day of a guess of when moonrise or moonset
713  might be
714  getMoonrise - If true, get moonrise time. If false, get moonset time
715  deltas - delta times, to convert from UTC to TT
716  site - properties of the site. Note: it is assumed that the field
717  site->refracPT will be set to 0.0 before calling this
718  routine, in order to calculate an unrefracted position of
719  the Moon (as per note below).
720  Output
721  topo - Topocentric position of Moon at approx. rise or set time
722 
723  Moonrise and set calculations assume a standard refraction at the horizon of
724  34 arcminutes, and a moon semi-diameter of 16 arcminutes. So the calculation
725  is based on the time at which the UNREFRACTED Moon position is at -50 arcmin.
726 - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - */
727 {
728  double ha1_rad; // Hour angle of Moon at time riseSetGuess_d(rad)
729  double dec_rad; // Declination of Moon (radian)
730  double ha2_rad; // Hour angle of Moon at horizon (radian)
731  double cosHa2; // Cos(Hour Angle) at horizon
732  double riseSetApprox_d;// Improved estimate of rise or set time
733 
734  moon_nrelTopocentric(risesetGuess_d, deltas, site, topo);
735  sky_siteAzElToHaDec(&topo->rectV, site, &ha1_rad, &dec_rad);
736 
737  /* Assuming the Moon's declination remains constant over the period, find
738  * out where dec circle intersects the Elevation = -50 arcminutes circle.
739  * Of course the declination does not remain constant, so this estimate of
740  * Hour Angle is approximate, but it will be a better estimate than
741  * ha1_rad. */
742  cosHa2 = (sin(degToRad(-50.0/60.0)) - sin(site->astLat_rad) * sin(dec_rad))
743  / (cos(site->astLat_rad) * cos(dec_rad));
744  /* If there is no intersection, the Moon either doesn't rise or doesn't set
745  on this day at this latitude. (In practice, does this ever happen?) */
746  if (fabs(cosHa2) > 1.0) {
747  riseSetApprox_d = 0.0;
748  } else {
749  if (getMoonrise) {
750  ha2_rad = -acos(cosHa2);
751  } else {
752  ha2_rad = acos(cosHa2);
753  }
754  /* Subtract difference in HA from Solar time (not sidereal time). If
755  the difference is large, there is a risk we will return the rise or
756  set time for the day before or the day after the one requested. Make
757  sure that doesn't happen first. The magic number of 1.03 is a little
758  fudge factor that seems to help this routine converge a bit faster.*/
759  if ((ha1_rad - ha2_rad) > PI) { risesetGuess_d += 1.0; }
760  if ((ha1_rad - ha2_rad) < -PI) { risesetGuess_d -= 1.0; }
761  riseSetApprox_d = risesetGuess_d - (ha1_rad - ha2_rad) / TWOPI * 1.03;
762  }
763  return riseSetApprox_d;
764 }
