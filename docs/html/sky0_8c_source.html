222 GLOBAL void sky0_nutationSpa(double t_cy, Sky0_Nut1980 *nut)
223 /*! Calculates the nutation in longitude and obliquity, according to the
224  algorithm set out in the NREL SPA document. This is a simplified version
225  of the IAU 1980 Nutation Theory. Calculates first the fundamental nutation
226  arguments, and then a series of terms. There are 106 terms in the full
227  series, but this routine uses only the largest 63 of those terms.
228  \param[in] t_cy centuries since J2000.0, TT timescale
229  \param[out] nut field \a nut->dPsi_rad - Nutation in longitude Δψ (radian)\n
230  field \a nut->dEps_rad - Nutation in obliquity Δε (radian)
231 
232  \par References:
233  Reda, I. and Andreas, A. (2003), "Solar Position Algorithm
234  for Solar Radiation Applications", National Renewable Energy Laboratory,
235  NREL publication no. NREL/TP-560-34302, section 3.4.
236  However, the theory on which it is based, and the full sequence of terms,
237  can be found in:\n
238  Final report of the IAU Working Group on Nutation,
239  chairman P.K.Seidelmann, 1980,
240  published in _Celestial Mechanics_, Vol 27, pp79-106, 1982.\n
241  also
242  Kaplan,G.H. _USNO circular no. 163_, pp A3-6, 1981.\n
243  _Supplement to the Astronomical Almanac_ 1984.
244 
245  \par When to call this function
246  The values calculated by this routine change only slowly. If using this
247  routine as part of tracking a celestial object, using values that were
248  calculated for the nearest hour will introduce almost imperceptible errors.
249  So it need be called only once per hour.
250 - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - */
251 {
252  /* Fundamental Nutation arguments at date */
253  double l; // l - mean anomaly of the Moon (radian) (usually called M?)
254  double lp; // l' - mean anomaly of the Sun (Earth) (radian) (M'?)
255  double om; /* Ω - longitude of the ascending node of the Moon's mean orbit
256  on the ecliptic, measured from the mean equinox of date
257  (radian) */
258  double d; // D - mean elongation of the Moon from the Sun (radian)
259  double f; // F - Mean longitude of the Moon (L) minus mean longitude of
260  // the Moon's node (radian), = L - Ω
261 
262  double psiSum_masx10; // Nutation in longitude (units - 0.1 mas)
263  double epsSum_masx10; // Nutation in obliquity (units - 0.1 mas)
264  int i;
265  double a_rad; // angle - summation of args (radian)
266 
267  REQUIRE_NOT_NULL(nut);
268 
269  // Calculate FUNDAMENTAL ARGUMENTS in the FK5 reference system
270 
271  // Mean elongation of the moon from the sun (called X0 in NREL SPA)
272  d = degToRad(297.85036 + t_cy * (445267.11148
273  + t_cy * (-0.0019142
274  + t_cy * (1.0/189474.0))));
275 
276  // Solar Mean Anomaly = Mean longitude of the sun minus mean longitude of
277  // the sun's perigee (called X1 in NREL SPA)
278  lp = degToRad(357.52772 + t_cy * (35999.05034
279  + t_cy * (-0.0001603
280  + t_cy * (-1.0/300000.0))));
281 
282  // Lunar Mean Anomaly = Mean longitude of the moon minus mean longitude of
283  // the moon 's perigee (called X2 in NREL SPA)
284  l = degToRad(134.96298 + t_cy * (477198.867398
285  + t_cy * (0.0086972
286  + t_cy * (1.0/56250.0))));
287 
288  // Mean longitude of the moon minus mean longitude of the moon's node
289  // (called X3 in NREL SPA and (mistakenly, I think) called the Moon's
290  // Argument of Latitude)
291  f = degToRad(93.27191 + t_cy * (483202.017538
292  + t_cy * (-0.0036825
293  + t_cy * (1.0/327270.0))));
294 
295  // Longitude of the mean ascending node of the lunar orbit on the
296  // ecliptic, measured from the mean equinox of date (X4 in NREL SPA)
297  om = degToRad(125.04452 + t_cy * (-1934.136261
298  + t_cy * (0.0020708
299  + t_cy * (1.0/450000.0))));
300 #if 0
301  printf("Nutation Args: d (X0) = %f°, lp (X1) = %f°, l (X2) = %f°\n"
302  " f (X3) = %f°, om (X4) = %f°\n",
303  radToDeg(d), radToDeg(lp), radToDeg(l), radToDeg(f), radToDeg(om));
304 #endif
305 
306  // Multiply through the table of nutation co-efficients and add up all the
307  // terms.
308  psiSum_masx10 = 0.0;
309  epsSum_masx10 = 0.0;
310  for (i = 0; i < Y_COUNT; i++) {
311  a_rad = d * Y_TERMS[i][0] + lp * Y_TERMS[i][1] + l * Y_TERMS[i][2]
312  + f * Y_TERMS[i][3] + om * Y_TERMS[i][4];
313  psiSum_masx10 += sin(a_rad) * (PE_TERMS[i][TERM_PSI_A]
314  + t_cy * PE_TERMS[i][TERM_PSI_B]);
315  epsSum_masx10 += cos(a_rad) * (PE_TERMS[i][TERM_EPS_C]
316  + t_cy * PE_TERMS[i][TERM_EPS_D]);
317  }
318 
319  // Convert results to radians
320  nut->dPsi_rad = psiSum_masx10 * MILLIARCSECx10_TO_RAD;
321  nut->dEps_rad = epsSum_masx10 * MILLIARCSECx10_TO_RAD;
322 }

326 GLOBAL void sky0_epsilonSpa(double t_cy, Sky0_Nut1980 *nut)
327 /*! Calculate the obliquity of the ecliptic and the equation of the equinoxes
328  \param[in] t_cy centuries since J2000.0, TT timescale
329  \param[in,out] nut [in] field \a nut->dPsi_rad - Nutation in longitude Δψ,
330  as returned by function sky0_nutationSpa()
331  (radian)\n
332  [in] field \a nut->dEps_rad - Nutation in obliquity Δε,
333  as returned by function sky0_nutationSpa()
334  (radian)\n
335  [out] field \a nut->eps0_rad - Mean obliquity of the
336  ecliptic ε0 (radian)\n
337  [out] field \a nut->eqEq_rad - Equation of the equinoxes
338  = Δψ * cos(ε0) (radian) Note: not seconds
339 
340  \par Reference
341  Reda, I. and Andreas, A. (2003), "Solar Position Algorithm
342  for Solar Radiation Applications", National Renewable Energy Laboratory,
343  NREL publication no. NREL/TP-560-34302, section 3.5.1
344 
345  \par When to call this function
346  The values calculated by this routine change only slowly. If using this
347  routine as part of tracking a celestial object, using values that were
348  calculated for the nearest hour will introduce almost imperceptible errors.
349  So it need be called only once per hour.
350 - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - */
351 {
352  double eps0_as;
353  double u = t_cy/100.0;
354 
355  REQUIRE_NOT_NULL(nut);
356 
357  eps0_as = 84381.448
358  + u * (-4680.93
359  + u * (-1.55
360  + u * (1999.25
361  + u * (-51.38
362  + u * (-249.67
363  + u * (-39.05
364  + u * (7.12
365  + u * (27.87
366  + u * (5.79
367  + u * 2.45)))))))));
368  nut->eps0_rad = arcsecToRad(eps0_as);
369  // Calculate the Equation of the Equinoxes in radian (not in seconds, which
370  // is more usually done)
371  nut->eqEq_rad = nut->dPsi_rad * cos(nut->eps0_rad + nut->dEps_rad);
372 }

376 GLOBAL double sky0_gmSiderealTimeSpa(double du)
377 /*! Calculate the Greenwich mean sidereal time using the algorithm from the NREL
378  SPA document. This is basically the IAU 1982 algorithm, but the various
379  constants have been scaled in degrees.
380  \returns Greenwich Mean Sidereal Time (radian)
381  \param[in] du days since J2000.0, UT1 timescale
382 
383  \par Reference
384  Reda, I. and Andreas, A. (2003), "Solar Position Algorithm
385  for Solar Radiation Applications", National Renewable Energy Laboratory,
386  NREL publication no. NREL/TP-560-34302, section 3.8.1 & 3.8.2
387 
388  \par When to call this function
389  If you are tracking a celestial object, you need not call this function
390  directly. So long as you call sky0_appToTirs() every time around your
391  control loop, it will call this function for you.
392 - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - */
393 {
394  static const double B0 = 280.46061837 * DEG2RAD; // 67310.54841 s
395  static const double B1 = 360.98564736629 * DEG2RAD; // 86636.55536791 s
396  static const double B2 = 0.000387933 * DEG2RAD; // 0.093104 s
397  static const double B3 = (-1.0 / 38710000.0) * DEG2RAD; // -6.2e-6 s
398 #if 0
399  static const double B0d = 4.89496121274; // 3579;
400  static const double B1d = 6.300388098984957;
401  static const double B2d = 6.77070812713916E-06;
402  static const double B3d = -4.50872966157151E-10;
403 #endif
404 
405  double gmst_rad; // Greenwich Mean Sidereal Time (radian)
406  double tu; // Julian centuries since J2000.0, UT1 timescale
407 
408  tu = du / JUL_CENT;
409  gmst_rad = B0 + B1 * du + (tu * tu * (B2 + (tu * B3)));
410 
411  return normalize(gmst_rad, TWOPI);
412 }

416 GLOBAL void sky0_appToTirs(const V3D_Vector *appV,
417  double j2kUT1_d,
418  double eqEq_rad,
419  V3D_Vector *terInterV)
420 /*! Convert a position in geocentric apparent coordinates to geocentric
421  coordinates in the Terrestrial Intermediate Reference System. This is the
422  first stage of converting apparent coordinates to topocentric coordinates.
423  The resulting vector depends upon the current rotational position of the
424  Earth. (For the second stage, to obtain topocentric coordinates, call
425  routine sky_siteTirsToTopo()).
426  \param[in] appV Position vector of apparent place
427  (unit vector in equatorial coordinates)
428  \param[in] j2kUT1_d days since J2000.0, UT1 timescale, as returned by function
429  sky_updateTimes() in the \a j2kUT1_d field of the
430  Sky_Times struct.
431  \param[in] eqEq_rad Equation of the equinoxes (radian), as returned by function
432  sky0_epsilonSpa() in the \a eqEq_rad field of the
433  Sky0_Nut1980 struct.
434 
435  \param[out] terInterV Position vector in Terrestrial Intermediate Ref System
436 
437  \par When to call this function
438  When you have the position of a celestial object expressed in Apparent
439  coordinates, use this function to convert it to Terrestrial coordinates at
440  the relevant rotational position of the earth. (Effectively you are
441  converting from Right Ascension and Declination to the negative of the
442  Greenwich Hour Angle and Declination.)
443  If you are running a control loop to enable continuous tracking
444  of this object, you will need to call this function (once) every time around
445  your control loop.
446  \par
447  Follow this function with a call to sky_siteTirsToTopo() to obtain the
448  object's position in topocentric coordinates at the observing site.
449 - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - */
450 {
451  double gast_rad; // Greenwich Apparent Sidereal Time (GAST)
452  V3D_Matrix earthRotM; // rotation matrix for current GAST
453 
454  REQUIRE_NOT_NULL(appV);
455  REQUIRE_NOT_NULL(terInterV);
456 
457  /* Get sidereal times */
458  gast_rad = sky0_gmSiderealTimeSpa(j2kUT1_d) + eqEq_rad;
459 
460  /* Create earth rotation matrix from the current Apparent Sidereal Time */
461  v3d_createRotationMatrix(&earthRotM, Zaxis, gast_rad);
462 
463  /* Convert apparent posn to posn in Terrestrial Intermediate Ref System */
464  v3d_multMxV(terInterV, &earthRotM, appV);
465 }
